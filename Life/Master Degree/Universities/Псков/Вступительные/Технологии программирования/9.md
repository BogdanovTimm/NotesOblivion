```cpp
// Read line-byline using functions from C:
FILE* file1;
file1 = fopen("newfile.txt", "r");
char arrayOfCharacters[100];
while(fgets(arrayOfCharacters, 100, file1)) {
    printf("%s", arrayOfCharacters);
}
fclose(file1);
return 0; // exit code
```
